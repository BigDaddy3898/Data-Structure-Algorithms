#include <iostream>

using namespace std;

inline int checkBigger(const int& a, const int& b)
{
	if (b >= a)
	{
		return b;
	}
	return a;
}
int main()
{
	int n;
	cin >> n;
	int* arr = new int [n];
	for (int i = 0; i < n; i++)
	{
		cin >> arr[i];
	}
	for (int i = 0; i < n; i++)
	{
		for (int j = i + 1; j < n - 1; j++)
		{
			if (arr[i] > arr[j])
			{
				int temp = arr[i];
				arr[i] = arr[j];
				arr[j] = temp;
			}
		}
	}
	int temp1 = arr[0] * arr[1];
	int temp2 = arr[n-1] * arr[n-2];
	cout << checkBigger(temp1, temp2) << "\n";

		delete[] arr;
	system("pause");
	return 0;
}
